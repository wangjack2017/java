#include"iostream"
int main()
{
	int y, m = 0, w;
	int i,j,n ;
	char a[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
	scanf_s("%d",&y);

	if (y % 4 == 0 && y % 100 != 0 || y % 400 == 0) {
		a[1] = 29;            
	}
	w = ((y - 1) * 365 + (y - 1) / 4 - (y - 1) / 100 + (y - 1) / 400 + 1) % 7;
	for (i = 1; i <= 12; i++) {
		printf("*********************");
		printf("%d 月\n",i);
		printf(" 日 一 二 三 四 五 六\n");
		for (j= 1; j<= w; j++) { 
			printf("   ");
			m++;
		}
		for (n = 1; n <= a[i - 1]; n++) {
			printf("%3d", n);
			m++;
			if (m % 7 == 0) {
				printf("\n");
				m = 0;
			}
		}
		printf("\n");
		w = (w + a[i - 1]) % 7;
		m = 0;

	}
	


	
	


	                                                                                 


	system("pause");
}
